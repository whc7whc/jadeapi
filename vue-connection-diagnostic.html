// 可能在以下檔案中： // src/config/api.js 或 src/utils/request.js 或 .env const API_BASE_URL = 'https://jadeapi-production.up.railway.app' // 或在 .env.production 檔案： VUE_APP_API_URL=https://jadeapi-production.up.railway.app

2. 檢查 Axios 配置

import axios from 'axios' const api = axios.create({ baseURL: 'https://jadeapi-production.up.railway.app', timeout: 10000, headers: { 'Content-Type': 'application/json' } })

3. 更新環境變數

// .env.production VUE_APP_API_URL=https://jadeapi-production.up.railway.app VUE_APP_ENV=production
